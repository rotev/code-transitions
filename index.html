describe('ItemPaneSidebarActions', () => {
              test('Convert to issue', async () => {
                const {wrapper, itemStore, statsClient} = setupEnvironment()
                const item = getDraftIssueModel(itemStore)
                jest.spyOn(statsClient, 'postStats')
            
                expect(item.contentType).toEqual('Issue')
                expect(statsClient.postStats).toHaveBeenCalledWith({
                  payload: {
                    memexProjectItemId: item.id,
                    name: 'draft_convert',
                    ui: 'item-pane'
                  }
                })
              })
